028 *

030*/

040}

101/**
102 * The class can render a custom html using lodash string templates.
103 *
104 * Inside the template you will have access to the <i>params</i> object
105 * which contains value, valueFormatted, ...
106 *
107 * Use This render if you want to render the data with a custom html.
108 *
109 * @see <a href = "https://lodash.com/docs/4.17.15#template">Lodash Templates</a>
110 *
111 * @author Hyyan Abo Fakher
112*/
113public class GxRendererCustomHTML extends GxRendererAbstract {
114/**
115 * String template which will be compiled in the client.<br>
116 * <pre>
117 * <code>
118 * <% = '<strong style = \'color:red\'>' + (params.valueFormatted || params.value) + '</strong>' %>
119 * </code>
120 * </pre>
121*/
122public BBjString Template!;
123/**
124 * {@inheritDoc}
125*/
126public static BBjString getCellRendererName() { }
127/**
128 * Construct new GxRendererCustomHTML
129 *
130 * @param BBjString template! The string template
131 *
132 * i.e
133 * <pre>
134 * <code>
135 * <% = '<strong style = \'color:red\'>' + (params.valueFormatted || '') + '</strong>' %>
136 * </code>
137 * </pre>
138*/
139public GxRendererCustomHTML(BBjString template!) { }
140/**
141 * {@inheritDoc}
142*/
143public JsonObject getAsJsonObject() { }
144}
