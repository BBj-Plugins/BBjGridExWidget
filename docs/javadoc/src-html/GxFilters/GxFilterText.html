128/**
129 * The class holds the shared ag filter options between all filter types. This is the base class for all ag filter types
130 *
131 * @author Hyyan Abo Fakher
132*/
133public class GxColumnFilterAG implements GxFilterInterface {
134/**
135 *Set to true to have the filter use an Apply button. If the Apply button is present, then the filter is only applied after the user hits the Apply button.
136*/
137public BBjNumber ApplyButton!;
138/**
139 * Set to true to have the filter use a Clear button. The Clear button will clear the details of the filter thus resetting it.
140*/
141public BBjNumber ClearButton!;
142/**
143 * Set to true to have the filter use a Reset button. The Reset button will clear the details of the filter and any active filters on that column.
144*/
145public BBjNumber RestButton!;
146/**
147 * The number of milliseconds to debounce by before applying the filter
148*/
149public BBjNumber DebounceMs!;
150/**
151 * If set to 'clear', then setting data into the grid will clear (reset) the filter If set to 'keep' then the grid will keep it's currently set filter. The default is 'clear'
152*/
153public BBjString NewRowsAction!;
154/**
155 * If true, the filter will only offer Condition 1.
156*/
157public BBjNumber SuppressAndOrCondition!;
158/**
159 * What Filter Options to present to the user.
160 * By default all supported options are presented
161 *
162 * @see GxColumnFilterTextFilterOptions
163 * @see GxColumnFilterNumberFilterOptions
164*/
165public JsonArray FilterOptions!;
166/**
167 * The default Filter Options to be selected.
168 * By default it is Equals
169*/
170public BBjString DefaultOption!;
171/**
172 * A constant which describes the NewRowsAction behavior
173 *
174 * @return BBjString <b>clear</b>
175*/
176public static BBjString NEW_ROWS_ACTION_CLEAR() { }
177/**
178 * A constant which describes the NewRowsAction behavior
179 *
180 * @return BBjString <b>keep</b>
181*/
182public static BBjString NEW_ROWS_ACTION_KEEP() { }
183/**
184 * {@inheritDoc}
185*/
186public JsonObject getAsJsonObject() { }
187/**
188 * {@inheritDoc}
189*/
190public boolean equals(GxFilterInterface filter!) { }
191}

261/**
262 * GxFilterSetFilter allow you to filter on set data, influenced by how filters work in Microsoft Excel
263 *
264 * <br><b><small>#Enterprise</small></b>
265 *
266 * @author Hyyan Abo Fakher
267*/
268public class GxFilterSetFilter extends GxColumnFilterAG {
269/**
270 * Provide a cell renderer for each value
271*/
272public GxRendererInterface CellRenderer!;
273/**
274 * The values to display in the filter. If this is not set, then the filter will takes its values
275 * from what is loaded in the table.
276 * Setting it allows you to set values where the value may not be present in the list
277 * (for example, if you want to show all states in America so that the user is not confused by missing states, even though states are missing from the data set in the grid)
278*/
279public JsonArray Values!;
280/**
281 * Set to true to have the values inside the set filter NOT refresh after values are changed inside the grid.
282*/
283public BBjNumber SuppressSyncValuesAfterDataChange!;
284/**
285 * Set to true to stop the filter from removing values that are no longer available (like what Excel does).
286*/
287public BBjNumber SuppressRemoveEntries!;
288/**
289 * If true, sorting will not be done on the set filter values. Use this if providing your own values and don't want them sorted as you are providing in the order you want.
290*/
291public BBjNumber SuppressSorting!;
292/**
293 * Set to true to hide the mini filter.
294*/
295public BBjNumber SuppressMiniFilter!;
296/**
297 * Set to true to remove the "select all" checkbox.
298*/
299public BBjNumber SuppressSelectAll!;
300/**
301 * {@inheritDoc}
302*/
303public static BBjString getFilterName() { }
304/**
305 * {@inheritDoc}
306*/
307public JsonObject getAsJsonObject() { }
308}
