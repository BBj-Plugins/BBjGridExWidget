158/**
159 * The class manages the grid sidebar and its toolpanels
160 *
161 * @author Hyyan Abo Fakher
162*/
163public class BBjGridExWidgetSidebar {
164/**
165 * The id of the default panel
166*/
167public BBjString DefaulToolpanel$;
168/**
169 * When true , the sidebar will be hidden on the first draw , visible otherwise.default is false
170*/
171public BBjNumber HiddenByDefault!;
172/**
173 * A map for the registered toolpanels
174*/
175public LinkedHashMap Toolpanels!;
176/**
177 * The BBjGridExWidget instance
178*/
179protected BBjGridExWidget Widget!;
180/**
181 * Disable default constructor
182*/
183private BBjGridExWidgetSidebar() { }
184/**
185 * Construct new sidebar
186 *
187 * @param BBjGridExWidget widget!
188*/
189public BBjGridExWidgetSidebar(BBjGridExWidget widget!) { }
190/**
191 * Set Visible
192 *
193 * Enable / Disable the grid sidebar
194*/
195public void setVisible(BBjNumber visible!) { }
196/**
197 * Open toolpanel
198 *
199 * Open the given toolpanel
200 *
201 * @param BBjString id$ the toolpanel id
202*/
203public void openToolpanel(BBjString id!) { }
204/**
205 * Close toolpanel
206 *
207 * Close the given toolpanel
208 *
209 * @param BBjString id$ the toolpanel id
210*/
211public void closeToolpanel(BBjString id!) { }
212/**
213 * Convert the sidebar into json object
214 *
215 * @returns JsonObject
216*/
217public JsonObject getAsJsonObject() { }
218}
