020*/

036public BBjString getLanguage() { }

044 * @param BBjString path$ the language file (JSON file)
045*/
046public void add(BBjString language$ , BBjString path$) { }
047/**

061/**

065 *

068public BBjString getLanguageKeys(BBjString language$) { }
