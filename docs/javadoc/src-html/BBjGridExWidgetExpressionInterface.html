069*/

072 * Construct a new BBjGridExWidgetExpressionNumbersFormatter
073 *
074 * @param BBjString mask! The numbers mask
075*/
076public BBjGridExWidgetExpressionNumbersFormatter(BBjString mask!) { }
077/**
078 * Construct a new BBjGridExWidgetExpressionNumbersFormatter
079 *
080 * @param BBjString mask! The numbers mask
081 * @param BBjString groupSeperator! The numbers group seperator to use
082*/
083public BBjGridExWidgetExpressionNumbersFormatter(BBjString mask! , BBjString groupSeperator!) { }
084/**
085 * Construct a new BBjGridExWidgetExpressionNumbersFormatter

087 * @param BBjString mask! The numbers mask

091public BBjGridExWidgetExpressionNumbersFormatter(BBjString mask! , BBjString groupSeperator! , BBjString decimalSeparator!) { }

093 * {@inheritDoc}
094*/
095public BBjString toString() { }
