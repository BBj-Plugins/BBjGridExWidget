009/**
010 * Column Filter Public Interface
011*/
012public interface BBjGridExWidgetFilterInterface {
013/**
014 * A constant which holds the client filter real name
015*/
016public static BBjString FILTER_NAME() { }
017/**
018 * Convert the filter defintion to Json Object
019 *
020 * @return JsonOobject The filter as JsonObject
021*/
022public JsonObject getAsJsonObject() { }
023}
024/**
025 * The class holds the shared ag filter options between all filter types. This is the base clasas for all filter types
026 *
027 * @author Hyyan Abo Fakher
028*/
029private class BBjGridExWidgetColumnFilterAG implements BBjGridExWidgetFilterInterface {
030/**
031 *Set to true to have the filter us an Apply button. If the Apply button is present, then the filter is only applied after the user hits the Apply button.
032*/
033public BBjNumber ApplyButton!;
034/**
035 * Set to true to have the filter us a Clear button. The Clear button will clear the details of the filter thus resetting it.
036*/
037public BBjNumber ClearButton!;
038/**
039 * The number of milliseconds to debounce by before applying the filter
040*/
041public BBjNumber DebounceMs!;
042/**
043 * If set to 'clear', then setting data into the grid will clear (reset) the filte If set to 'keep' then the grid will keep it's currently set filter. The default is 'clear'
044*/
045public BBjString NewRowsAction!;
046/**
047 * If true, the filter will only offer Condition 1.
048*/
049public BBjNumber SuppressAndOrCondition!;
050/**
051 * A constant which descripe the NewRowsAction behavior
052 *
053 * @return BBjString <b>clear</b>
054*/
055public static BBjString NEW_ROWS_ACTION_CLEAR() { }
056/**
057 * A constant which descripe the NewRowsAction behavior
058 *
059 * @return BBjString <b>keep</b>
060*/
061public static BBjString NEW_ROWS_ACTION_KEEP() { }
062/**
063 * {@inheritDoc}
064*/
065public JsonObject getAsJsonObject() { }
066}

088/**
089 * The class represents the defualt ag-grid number filter
090 *
091 * @author Hyyan Abo Fakher
092*/
093public class BBjGridExWidgetFilterAGNumber extends BBjGridExWidgetColumnFilterAG {
094/**
095 * If true then doing 'inRange' filter option will include values equal to the start and end of the range.
096*/
097public BBjNumber InRangeInclusive!;
098/**
099 * If true then blank (null or undefined) values will pass the 'equals' filter option.
100*/
101public BBjNumber IncludeBlanksInEquals!;
102/**
103 * If true then blank (null or undefined) values will pass the 'lessThan' and 'lessThanOrEqual' filter options.
104*/
105public BBjNumber IncludeBlanksInLessThan!;
106/**
107 * If true then blank (null or undefined) values will pass the 'greaterThan' and 'greaterThanOrEqual' filter options.
108*/
109public BBjNumber IncludeBlanksInGreaterThan!;
110/**
111 * {@inheritDoc}
112*/
113public static BBjString FILTER_NAME() { }
114/**
115 * @override
116 *
117 * {@inheritDoc}
118*/
119public JsonObject getAsJsonObject() { }
120}
